🔗 เชื่อมต่อ Google Sheet
วาง URL ของ Apps Script เพื่อบันทึกและดึงข้อมูลอัตโนมัติ
📖 ขั้นตอนตั้งค่า (ทำครั้งเดียว)
1
เปิด sheets.google.com → สร้าง Spreadsheet ใหม่ → ตั้งชื่อตามต้องการ
2
คลิกเมนู Extensions → Apps Script
3
ลบโค้ดเดิมทิ้ง → วางโค้ดด้านล่าง → กด 💾 Save
📋 คัดลอกโค้ด Apps Script
function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const d = JSON.parse(e.postData.contents); let sh = ss.getSheetByName(d.sheet); if (!sh) { sh = ss.insertSheet(d.sheet); sh.appendRow(d.headers); } else if (sh.getLastRow() === 0) { sh.appendRow(d.headers); } d.rows.forEach(r => sh.appendRow(r)); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput('API OK'); }

5
