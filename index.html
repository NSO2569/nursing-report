🔗 เชื่อมต่อ Google Sheet
วาง URL ของ Apps Script เพื่อบันทึกและดึงข้อมูลอัตโนมัติ
📖 ขั้นตอนตั้งค่า (ทำครั้งเดียว)
1
เปิด sheets.google.com → สร้าง Spreadsheet ใหม่ → ตั้งชื่อตามต้องการ
2
คลิกเมนู Extensions → Apps Script
3
ลบโค้ดเดิมทิ้ง → วางโค้ดด้านล่าง → กด 💾 Save
📋 คัดลอกโค้ด Apps Script
function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let body = e.postData ? e.postData.contents : ''; // รองรับ GET fallback ที่ส่ง payload ใน query string if (!body && e.parameter && e.parameter.payload) body = decodeURIComponent(e.parameter.payload); const d = JSON.parse(body); let sh = ss.getSheetByName(d.sheet); if (!sh) { sh = ss.insertSheet(d.sheet); sh.appendRow(d.headers); } else if (sh.getLastRow() === 0) { sh.appendRow(d.headers); } d.rows.forEach(r => sh.appendRow(r)); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const p = e.parameter; // รองรับ POST ผ่าน GET (มือถือ fallback) if (p.payload) return doPost(e); const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = p.sheet || 'ward_census'; const date = p.date || ''; const month = p.month || ''; const sh = ss.getSheetByName(sheet); if (!sh) return ContentService .createTextOutput(JSON.stringify({ok:false,rows:[]})) .setMimeType(ContentService.MimeType.JSON); const data = sh.getDataRange().getValues(); if (data.length < 2) return ContentService .createTextOutput(JSON.stringify({ok:true,headers:[],rows:[]})) .setMimeType(ContentService.MimeType.JSON); const headers = data[0]; let rows = data.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); if (date) rows = rows.filter(r => r['วันที่'] === date); if (month) rows = rows.filter(r => String(r['วันที่']).includes(month)); return ContentService .createTextOutput(JSON.stringify({ok:true, headers, rows})) .setMimeType(ContentService.MimeType.JSON); }
