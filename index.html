// ══════════════════════════════════════ // Apps Script — รองรับทุก device + CORS // ══════════════════════════════════════ function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let body = ''; try { body = e.postData.contents; } catch(ex) {} if (!body && e.parameter && e.parameter.payload) body = decodeURIComponent(e.parameter.payload); const d = JSON.parse(body); let sh = ss.getSheetByName(d.sheet); if (!sh) { sh = ss.insertSheet(d.sheet); sh.appendRow(d.headers); } else if (sh.getLastRow() === 0) { sh.appendRow(d.headers); } d.rows.forEach(r => sh.appendRow(r)); return buildResponse(JSON.stringify({ok:true})); } function doGet(e) { const p = e.parameter || {}; if (p.payload) return doPost(e); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (p.all === '1') { const sheetNames = ['ward_census','er_census','cl_census','or_census','ws_census','opd_census','special_clinic']; const result = {}; sheetNames.forEach(name => { result[name] = getSheetRows(ss, name, p.date, p.month); }); return buildResponse(JSON.stringify({ok:true, data:result})); } const rows = getSheetRows(ss, p.sheet || 'ward_census', p.date, p.month); return buildResponse(JSON.stringify({ok:true, rows})); } function getSheetRows(ss, sheetName, dateFilter, monthFilter) { const sh = ss.getSheetByName(sheetName); if (!sh) return []; const data = sh.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; let rows = data.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => { if(h) obj[String(h)] = r[i]; }); return obj; }); if (dateFilter) rows = rows.filter(r => r['วันที่'] == dateFilter); if (monthFilter) rows = rows.filter(r => String(r['วันที่']).includes(monthFilter)); return rows; } function buildResponse(json) { return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
